package org.example.findindex;

import java.util.Scanner;

public class Binary {
        public static void main(String[] args) {
            Scanner input = new Scanner(System.in);
            int n = input.nextInt();
            int[]arr = new int [n];
            for (int i = 0; i < n; i++){
                arr [1] = input.nextInt();
            }
            int c = input.nextInt();
            for (int i = 0; i < n; i++){
                if(arr[i] == c){
                    System.out.println(i);
                }
            }
        }

    }
